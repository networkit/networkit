38 Clustering EnsembleMultilevel::run(Graph& G) {
39  // DEBUG
40  INFO("STARTING EnsembleClusterer on G=" << G.toString());
41  // DEBUG
42 
43 
44  // sub-algorithms
45  ClusterContracter contract;
46  ClusteringProjector project;
47 
48  // hierarchies
49  std::vector<Graph> graph; // hierarchy of graphs G^{i}
50  std::vector<Clustering> clustering; // hierarchy of core clusterings \zeta^{i}
51  std::vector<NodeMap<node> > map; // hierarchy of maps M^{i->i+1}
52  std::vector<double> quality; // hierarchy of clustering quality values q^{i} = q(\zeta^{i}, G^{0})
53 
54  // other data collections
55  std::vector<Clustering> baseClustering(baseClusterers.size(), Clustering(0)); // collection of base clusterings - fill with empty clustering
56 
57  // DEBUG
58  GraphIO graphio;
59  // DEBUG
60 
61  bool repeat;
62 
63  graph.push_back(G); // store G^{0}
64  Clustering empty(0);
65 
66  Clustering best(0); // best clustering seen so far
67  double bestQuality = -2.0; // quality of best clustering
68  int bestLevel = -1; // level of best clustering
69  Modularity modularity;
70 
71  int iterationsWithoutImprovement = 0;
72  const int maxIterationsWithoutImprovement = 3;
73 
74  do {
75  h += 1; // increment iteration/hierarchy counter
76  INFO("EnsembleClusterer *** ITERATION " << h << " ***");
77 
78  // *** base clusterers calculate base clusterings ***
79  #pragma omp parallel for
80  for (int b = 0; b < baseClusterers.size(); b += 1) {
81  try {
82  baseClustering.at(b) = baseClusterers.at(b)->run(graph.at(h));
83  // DEBUG
84  DEBUG("created base clustering: k=" << baseClustering.at(b).numberOfClusters());
85  if (baseClustering.at(b).isOneClustering(graph.at(h))) {
86  WARN("base clusterer created 1-clustering");
87  }
88  // DEBUG
89  } catch (...) {
90  ERROR("base clusterer failed with exception.");
91  throw std::runtime_error("base clusterer failed.");
92  }
93  }
94 
95  for (int b = 0; b < baseClusterers.size(); b += 1) {
96  double qual = modularity.getQuality(baseClustering[b], graph.at(h));
97  INFO("base clustering quality: " << qual);
98  if (qual > bestQuality) {
99  bestQuality = qual;
100  best = baseClustering[b];
101  bestLevel = h;
102  }
103  }
104 
105  // ANALYSIS
106  if (CALC_DISSIMILARITY) {
107  JaccardMeasure dm;
108  double dissimilaritySum = 0.0;
109  for (int b = 0; b < baseClustering.size(); b += 1) {
110  for (int c = b + 1; c < baseClustering.size(); c += 1) {
111  double d = dm.getDissimilarity(graph.at(h), baseClustering.at(b), baseClustering.at(c));
112  dissimilaritySum += d;
113  INFO("dm " << b << " <-> " << c << ": " << d);
114  }
115  }
116  double avgDissimilarity = dissimilaritySum / (baseClustering.size() * (baseClustering.size() - 1) / 2.0);
117  std::cout << "[INFO] avg. base clustering dissimilarity for h=" << this->h << ": " << avgDissimilarity << std::endl;
118  }
119  //
120 
121  // *** overlap clusters to create core clustering ***
122  INFO("[BEGIN] finding core clustering");
123  DEBUG("graph at pos " << h << ": " << graph.at(h).toString());
124  clustering.push_back(this->overlap->run(graph.at(h), baseClustering));
125  DEBUG("created core clustering: k=" << clustering.at(h).numberOfClusters());
126  INFO("[DONE] finding core clustering");
127 
128  if (h == 0) { // first iteration
129  // *** calculate quality of first core clustering with respect to first graph ***
130  quality.push_back(this->qm->getQuality(clustering.at(h), graph.at(h)));
131  DEBUG("pushed quality: " << quality.back());
132 
133  if (quality.back() > bestQuality) {
134  bestQuality = quality.back();
135  best = clustering.back();
136  bestLevel = h;
137  }
138 
139  // new graph created => repeat
140  repeat = true;
141 
142  // *** contract the graph according to core clustering **
143  INFO("[BEGIN] contracting graph");
144  auto con = contract.run(graph.at(h), clustering.at(h)); // returns pair (G^{i+1}, M^{i->i+1})
145  graph.push_back(con.first); // store G^{i+1}
146  map.push_back(con.second); // store M^{i->i+1}
147 
148  //DEBUG
149  DEBUG("contracted graph G^" << (h+1) << " created: " << graph.back().toString());
150  //DEBUG
151 
152  } else { // other iterations
153  quality.push_back(this->qm->getQuality(clustering.at(h), graph.at(h)));
154  INFO("new quality: " << quality.back());
155  std::cout << "[INFO] new quality: " << quality.back() << std::endl;
156 
157  if (quality.back() > bestQuality) {
158  bestQuality = quality.back();
159  best = clustering.at(h);
160  bestLevel = h;
161  }
162 
163  // *** test if new core clustering is better than previous one **
164  if (quality.at(h) > quality.at(h-1)) {
165  DEBUG("quality[" << h << "] = " << quality.at(h) << " > quality[" << (h-1) << "] = " << quality.at(h-1));
166 
167  // better quality => repeat
168  repeat = true;
169  iterationsWithoutImprovement = 0;
170  } else {
171  DEBUG("quality[" << h << "] = " << quality.at(h) << " <= quality[" << (h-1) << "] = " << quality.at(h-1));
172 
173  ++iterationsWithoutImprovement;
174  if (iterationsWithoutImprovement
175  >= maxIterationsWithoutImprovement) {
176  // new core clustering is not better for some time => do not contract according to new core clustering and do not repeat
177  repeat = false;
178  }
179  }
180 
181  if (repeat) {
182  INFO("[BEGIN] contracting graph");
183  auto con = contract.run(graph.at(h), clustering.at(h)); // returns pair (G^{i+1}, M^{i->i+1})
184  graph.push_back(con.first); // store G^{i+1}
185  map.push_back(con.second); // store M^{i->i+1}
186  INFO("[DONE] contracting graph");
187  }
188  }
189 
190  } while (repeat);
191 
192  Clustering zetaFine(0);
193  bool bestInCoreGroup = false;
194  if (bestLevel > 0) {
195  Clustering zetaCoarse = this->finalClusterer->run(graph.at(h)); // TODO: check: index correct?
196  double mod = modularity.getQuality(zetaCoarse, graph.at(h));
197 
198  if (mod > bestQuality) {
199  bestQuality = mod;
200  best = zetaCoarse;
201  bestLevel = h;
202  bestInCoreGroup = true;
203  }
204  zetaFine = project.projectBackToFinest(best, map, graph.at(bestLevel));
205 
206  } else {
207  zetaFine = best;
208  }
209 
210  INFO("Best clustering was found on level " << bestLevel << ", quality: "
211  << modularity.getQuality(zetaFine, G) << " vs " << bestQuality << " vs "
212  << modularity.getQuality(best, graph.at(bestLevel)) << ", in core group? " << bestInCoreGroup);
213 
214  return zetaFine;
215 }
